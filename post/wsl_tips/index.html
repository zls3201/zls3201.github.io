WSL 常用命令

  • 技术
  • 2021-05-10 
# 安装
dism.exe /online /enable-feature /featurename:Microsoft-Windows-Subsystem-Linux /all /norestart
dism.exe /online /enable-feature /featurename:VirtualMachinePlatform /all /norestart

Enable-WindowsOptionalFeature -Online -FeatureName Microsoft-Windows-Subsystem-Linux

wsl --set-default-version 2

# 查看实例
wsl -l -v
LxRunOffline l

# 查看实例IP
wsl hostname -I

# 安装Ubuntu-16.04.2
LxRunOffline i -n Ubuntu-16.04.2 -d f:\wsl\Ubuntu-16.04.2 -f "F:\wsl\16.04.2-server-cloudimg-amd64-root.tar.gz" -s

wsl --set-version Ubuntu-16.04.2 2
wsl --set-version Ubuntu-20.04 2

主机查看文件
type \\wsl$\zls-wsl\root\.profile

# 注册
LxRunOffline rg -n Ubuntu-20.04 -d f:\wsl\Ubuntu-20.04
LxRunOffline rg -n Ubuntu-16.04.2 f:\wsl\Ubuntu-16.04.2

# 删除注册（保留文件）
LxRunOffline ur -n zls-wsl

# 卸载并删除文件！！！
LxRunOffline ui -n zls-wsl

# 修改WSL安装路径
LxRunOffline move -n Ubuntu-20.04 -d f:\wsl\Ubuntu-20.04

# 查看
LxRunOffline get-dir -n Ubuntu-20.04

# 导出
wsl --export Ubuntu-20.04  ubuntu20.04.tar.gz

wsl --import <Image Name> <Directory where you want to store the imported image> <Directory where the exported .tar file exists>
